## 二、购物车模块接口

### 2.1 新增商品到购物车

**接口地址**: `POST /oms/cart/add`

**接口描述**: 将商品添加到购物车，如果已存在则累加数量

**请求头**:
```
Authorization: Bearer {token}
Content-Type: application/json
```

**请求参数**:

| 参数名 | 类型 | 必填 | 说明 | 示例 |
|--------|------|------|------|------|
| skuId | Long | 是 | SKU ID | 1 |
| title | String | 是 | 商品标题 | 小米11 Ultra黑色512G |
| mainPicture | String | 否 | 商品主图URL | http://xxx.jpg |
| price | BigDecimal | 是 | 商品单价 | 6999.99 |
| quantity | Integer | 是 | 购买数量 | 1 |

**请求示例**:
```json
{
  "skuId": 1,
  "title": "小米11 Ultra黑色512G",
  "mainPicture": "http://xxx.jpg",
  "price": 6999.99,
  "quantity": 1
}
```

**响应参数**:

| 参数名 | 类型 | 说明 |
|--------|------|------|
| state | Integer | 状态码，200成功 |
| message | String | 提示信息 |
| data | Object | 返回数据(null) |

**响应示例**:
```json
{
  "state": 200,
  "message": "新增sku操作完成",
  "data": null
}
```

### 5.1 购物车页面

```vue
<template>
  <div class="cart-page">
    <!-- 购物车列表 -->
    <nut-checkbox-group v-model="selectedIds" @change="handleSelectChange">
      <nut-swipe v-for="item in cartList" :key="item.id">
        <div class="cart-item">
          <nut-checkbox :label="item.id" />
          <nut-image :src="item.mainPicture" width="80" height="80" fit="cover" />
          <div class="item-info">
            <div class="title">{{ item.title }}</div>
            <div class="price">¥{{ item.price }}</div>
            <nut-input-number 
              v-model="item.quantity" 
              :min="1" 
              :max="99" 
              @change="updateQuantity(item)" 
            />
          </div>
        </div>
        <template #right>
          <nut-button type="danger" @click="deleteItem(item.id)">删除</nut-button>
        </template>
      </nut-swipe>
    </nut-checkbox-group>

    <!-- 空状态 -->
    <nut-empty v-if="cartList.length === 0" description="购物车空空如也" />

    <!-- 底部结算栏 -->
    <div class="cart-footer" v-if="cartList.length > 0">
      <nut-checkbox v-model="allSelected" @change="handleSelectAll">全选</nut-checkbox>
      <div class="total">
        合计: <span class="price">¥{{ totalPrice }}</span>
      </div>
      <nut-button type="primary" @click="goToSettle" :disabled="selectedIds.length === 0">
        结算({{ selectedIds.length }})
      </nut-button>
    </div>
  </div>
</template>

<script setup>
import { ref, computed, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import request from '@/utils/request'

const router = useRouter()
const cartList = ref([])
const selectedIds = ref([])
const allSelected = ref(false)

// 获取购物车列表
const fetchCartList = async () => {
  try {
    const res = await request.get('/oms/cart/list', {
      params: { page: 1, pageSize: 100 }
    })
    cartList.value = res.data.list || []
  } catch (error) {
    console.error('获取购物车失败:', error)
  }
}

// 修改数量
const updateQuantity = async (item) => {
  try {
    await request.post('/oms/cart/update/quantity', {
      id: item.id,
      quantity: item.quantity
    })
  } catch (error) {
    console.error('修改数量失败:', error)
  }
}

// 删除单个商品
const deleteItem = async (id) => {
  try {
    await request.post('/oms/cart/delete', null, {
      params: { ids: id }
    })
    fetchCartList()
  } catch (error) {
    console.error('删除失败:', error)
  }
}

// 全选/取消全选
const handleSelectAll = (val) => {
  if (val) {
    selectedIds.value = cartList.value.map(item => item.id)
  } else {
    selectedIds.value = []
  }
}

// 计算总价
const totalPrice = computed(() => {
  return cartList.value
    .filter(item => selectedIds.value.includes(item.id))
    .reduce((sum, item) => sum + item.price * item.quantity, 0)
    .toFixed(2)
})

// 去结算
const goToSettle = () => {
  router.push({
    path: '/order/settle',
    query: { cartIds: selectedIds.value.join(',') }
  })
}

onMounted(() => {
  fetchCartList()
})
</script>
```

```vue
import { ref, computed, onMounted } from 'vue'

const router = useRouter()

const orderItems = ref([])

  if (!selectedAddress.value) return ''
  const addr = selectedAddress.value
  return `${addr.provinceName}${addr.cityName}${addr.districtName}${addr.detailedAddress}`

// 商品总价
const totalPrice = computed(() => {
  return orderItems.value
    .reduce((sum, item) => sum + item.price * item.quantity, 0)
    .toFixed(2)
})

// 实付金额
const actualPay = computed(() => {
  return (parseFloat(totalPrice.value) + freight.value - discount.value).toFixed(2)
})

  selectedAddress.value = addr
  showAddressPicker.value = false
}

// 提交订单
const submitOrder = async () => {
  if (!selectedAddress.value) {
    return alert('请选择收货地址')
  }
  if (orderItems.value.length === 0) {
    return alert('请选择商品')
  }

  submitting.value = true
  try {
    const addr = selectedAddress.value
    const orderData = {
      contactName: addr.contactName,
      mobilePhone: addr.mobilePhone,
      telephone: addr.telephone || '',
      provinceCode: addr.provinceCode,
      provinceName: addr.provinceName,
      cityCode: addr.cityCode,
      cityName: addr.cityName,
      districtCode: addr.districtCode,
      districtName: addr.districtName,
      streetCode: addr.streetCode || '',
      streetName: addr.streetName || '',
      detailedAddress: addr.detailedAddress,
      tag: addr.tag || '',
      paymentType: paymentType.value,
      amountOfOriginalPrice: parseFloat(totalPrice.value),
      amountOfFreight: freight.value,
      amountOfDiscount: discount.value,
      orderItems: orderItems.value.map(item => ({
        skuId: item.skuId,
        title: item.title,
        barCode: item.barCode || '',
        data: item.data || '[]',
        mainPicture: item.mainPicture,
        price: item.price,
        quantity: item.quantity
      }))
    }

      query: {

        amount: res.data.payAmount

    })
  } catch (error) {
    console.error('下单失败:', error)
    alert(error.response?.data?.message || '下单失败')
  } finally {
    submitting.value = false
  }
}
```

## 六、接口调用流程

### 6.1 购物车流程
```
1. 用户点击"加入购物车" → POST /oms/cart/add
2. 查看购物车 → GET /oms/cart/list
3. 修改数量 → POST /oms/cart/update/quantity
4. 删除商品 → POST /oms/cart/delete
5. 选择商品 → 前端记录选中的cartIds
6. 点击结算 → 跳转到结算页
```

### 6.2 订单流程
```
1. 结算页获取选中商品 → GET /oms/cart/list (前端筛选)
2. 选择收货地址 → 前端处理
3. 提交订单 → POST /oms/order/add
4. 订单列表 → GET /oms/order/list
5. 订单详情 → GET /oms/order/detail
6. 取消订单 → POST /oms/order/update/state (state=2)
7. 确认收货 → POST /oms/order/update/state (state=4)
```
